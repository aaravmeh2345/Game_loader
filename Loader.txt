# Hardcoded credentials jo code ke andar hi save hain
VALID_USER = "admin"
VALID_KEY = "LocalTest123"

def check_login():
    print("=== LOCAL LOGIN SYSTEM ===")
    username = input("Enter Username: ")
    user_key = input("Enter Access Key: ")
    
    # Bina kisi server ke, local condition check
    if username == VALID_USER and user_key == VALID_KEY:
        print("\n[+] Login Successful! Tool loading...")
        start_main_program()
    else:
        print("\n[-] Invalid Credentials! Access Denied.")

def start_main_program():
    print("Aapka main feature ya tool yahan se run hoga.")

# Script ko start karna
if __name__ == "__main__":
    check_login()
